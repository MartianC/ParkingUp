        private void LoadData(string dataStr)
		{
	        string[] datas = dataStr.Split('\r');
	        for (int i = 5; i < datas.Length; i++)
	        {//数据从第五行开始
		        try
		        {
			        var item = new #DATANAME#Data(datas[i]);
			        Data[item.Id] = item;
		        }
		        catch (Exception e)
		        {
			        GameDebug.Log("#DATANAME#Data 初始化错误 at line:" + i);
		        }
	        }
		}
